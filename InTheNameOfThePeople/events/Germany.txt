country_event = {
	id = germany.1
	title = germany.1.t
	desc = germany.1.d

	is_triggered_only =  yes
	
	option = { 
		name = germany.1.a

		set_global_flag = german_civil_war
		declare_war_on = {
			target = PRS
			type = annex_everything
		}
	}
}
