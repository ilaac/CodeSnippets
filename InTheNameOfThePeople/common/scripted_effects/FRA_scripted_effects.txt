plan_alpha_decisions = {
	
	hidden_effect = {
		for_each_scope_loop = {
			array = global.countries 
			IF = {
				limit = { 
					any_state = { 
							is_core_of = FRA
							is_owned_by = FRA
						OWNER = {
						  tag = FRA 
						}
					}
				}

				every_state = {				
					limit = {
					is_core_of = FRA
					is_owned_by = FRA
						NOT = { 
							is_in_array = {
								array = FRA.plan_alpha_decisions
								value = THIS 
							}
						}
					}
					add_to_array = { FRA.plan_alpha_decisions = THIS }
				}
			}
			
		}
	}
}
integrate_eurasia = {
	
	hidden_effect = {
		for_each_scope_loop = {
			array = global.countries 
			IF = {
				limit = { 
					any_state = { 
							is_claimed_by = FRA
						OWNER = {
						  tag = FRA 
						}
					}
				}

				every_state = {				
					limit = {
					is_claimed_by = FRA
					OR = {
						is_on_continent = europe
						is_on_continent = asia
					}
						NOT = { 
							is_in_array = {
								array = FRA.integrate_eurasia
								value = THIS 
							}
						}
					}
					add_to_array = { FRA.integrate_eurasia = THIS }
				}
			}
			
		}
	}
}

increase_state_category2x = {
	if = {
		limit = {
			has_state_category = tiny_island
		}
		set_state_category = small_island
		effect_tooltip = {
			add_extra_state_shared_building_slots = 1
		}
	}

	if = {
		limit = {
			OR = {
				has_state_category = wasteland
				has_state_category = enclave
			}
		}
		set_state_category = pastoral
		effect_tooltip = {
			add_extra_state_shared_building_slots = 1
		}
	}
	else_if = {
		limit = {
			has_state_category = pastoral
		}
		set_state_category = small_town
		effect_tooltip = {
			add_extra_state_shared_building_slots = 2
		}
	}
	else_if = {
		limit = {
			has_state_category = rural
		}
		set_state_category = town
		effect_tooltip = {
			add_extra_state_shared_building_slots = 2
		}
	}
	else_if = {
		limit = {
			has_state_category = town
		}
		set_state_category = city
		effect_tooltip = {
			add_extra_state_shared_building_slots = 2
		}
	}
	else_if = {
		limit = {
			has_state_category = large_town
		}
		set_state_category = large_city
		effect_tooltip = {
			add_extra_state_shared_building_slots = 2
		}
	}
	else_if = {
		limit = {
			has_state_category = city
		}
		set_state_category = large_city
		effect_tooltip = {
			add_extra_state_shared_building_slots = 2
		}
	}

	else_if = {
		limit = {
			has_state_category = large_city
		}
		set_state_category = metropolis
		effect_tooltip = {
			add_extra_state_shared_building_slots = 2
		}
	}

	else_if = {
		limit = {
			has_state_category = metropolis
		} 
		set_state_category = megalopolis
		effect_tooltip = {
			add_extra_state_shared_building_slots = 2
		}
	}

}
